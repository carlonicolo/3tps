# Comandi Linux

pwd                 -> Restituisce il path corrente
ls                  -> Lista file e cartelle
ls -a               -> Lista di tutte le cartelle ed i file (anche quelli nascosti)
mkdir [nomeCartella]              -> Crea una cartella
cd Home/                  ->
cd ..               -> Ritorna indietro di una cartella
cat [file.txt]                -> Visualizzare il contenuto di file.txt
rm [file.txt]                 ->
rm -R [NomeCartella]               ->
nano nomeFile.txt   -> crea un file o modifica un file
mv nomeFile.txt nomeFile2.txt    -> Rinomina il file nomeFile.txt in nomeFile2.txt
mv nomeFile.txt /dev/null        -> Elimina definitivamente il file nomeFile.txt
cat ex1.txt
head ex1.txt
more ex1.txt
less ex1.txt
tail ex1.txt
